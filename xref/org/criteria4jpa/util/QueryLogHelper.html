39

41        while (parameterIterator.hasNext()) {
42

44          builder.append( objectToString( parameterIterator.next() ) );

49          }
50          
51        }
52

54        builder.append(" }");
55        
56      }
57

63     * Helper method to create a string representation of an object.
64     * 
65     * @param obj the object to print
66     * @return a string representation of the object

68    private static String objectToString(Object obj) {
69  
70      // handle 'null'
71      if(obj == null) {
72        return "null";
73      }
74      
75      // enclose Strings in quotes
76      if(obj instanceof CharSequence) {

78      }
79  
80      // use toString() for all other objects
81      return obj.toString();
