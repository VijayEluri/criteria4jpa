14    /**
15     * Method to create unique aliases for JPA identifiers. Each alias
16     * consists of two parts. The <i>basename</i> should be a human
17     * readable string that helps to identify which entity is referred. 
18     * The <i>counter</i> is unique number that is added to the
19     * alias to make the alias unique. It is recommended to use each
20     * counter value one once independently from the basename used.
21     * 
22     * @param basename a base name that will be used for alias creation
23     * @param counter A unique number
24     * @param fallback An optional fallback name
25     * @return the generated JPQL alias
26     */
27    public static String generateAlias(String basename, int counter, String fallback) {
28      
29      // build trimmed and lower case version of the base name
30      String lowerCaseBaseName = basename.trim().toLowerCase(Locale.ENGLISH);
31  
32      // use only simple characters for the alias
33      StringBuilder builder = new StringBuilder();
34      for( char c : lowerCaseBaseName.toCharArray() ) {
35        if( c >= 'a' && c <= 'z' ) {
36          builder.append( c ); 
37        }
38      }
39      
40      // add some default keyword, if builder is still empty
41      if( builder.length() == 0 ) {
42        if(fallback != null && fallback.length() > 0) {
43          builder.append( fallback );
44        } else {
45          builder.append( "default" );
46        }
47      }
48      
49      // finally append the counter to get uniqueness
50      builder.append( counter );
51      
52      // return the result
53      return builder.toString();
54    }

56    /**
57     * Returns the last component of a dot-separated path.
58     * 
59     * @param path relative or absolute path expression
60     * @return last component of path expression
61     */
62    public static String getLastPathComponent(String path) {
63      
64      // find last '.' character
65      int pos = path.lastIndexOf('.');
66      
67      // return the last path component
68      // or the complete path, if no '.' chars were found
69      return pos >= 0 ? path.substring(pos + 1) : path;
70    }
