29    public InExpression(String relativePath, Object[] values) {

35     * @see org.criteria4jpa.criterion.Criterion#toQueryString(org.criteria4jpa.Criteria, org.criteria4jpa.impl.CriteriaQueryBuilder)

37    public String toQueryString(Criteria criteria, CriteriaQueryBuilder queryBuilder) {
38  
39      // query builder
40      StringBuilder builder = new StringBuilder();
41  
42      // build the basic query
43      builder.append( queryBuilder.getAbsolutePath(criteria, relativePath) );
44      builder.append( " IN (");
45      
46      // We must add each value as a parameter, because not all JPA
47      // implementations allow lists or arrays as parameters.
48      if( values != null ) {
49        for( int i = 0; i < values.length; i++) {
50          builder.append( queryBuilder.createPositionalParameter() );
51          if( i < values.length - 1 ) {
52            builder.append(',');
53          }
54        }
55      }
56      builder.append(")");
57      
58      // return result
59      return builder.toString();
60    }
