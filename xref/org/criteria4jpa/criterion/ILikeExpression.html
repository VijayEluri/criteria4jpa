18  public class ILikeExpression implements Criterion {

21    private final String value;

23    /**
24     * Creates a case-insensitive LIKE expression 
25     * 
26     * @param relativePath relative path of a string field.
27     * @param value a value to check the field against
28     */
29    public ILikeExpression(String relativePath, String value) {
30      this.relativePath = relativePath;
31      this.value = value;
32    }

34    /*
35     * @see org.criteria4jpa.criterion.Criterion#toQueryString(org.criteria4jpa.Criteria, org.criteria4jpa.impl.CriteriaQueryBuilder)
36     */
37    public String toQueryString(Criteria criteria, CriteriaQueryBuilder queryBuilder) {
38      StringBuilder builder = new StringBuilder();
39      builder.append( "LOWER(" );
40      builder.append( queryBuilder.getAbsolutePath(criteria, relativePath) );
41      builder.append( ") LIKE " );
42      builder.append( queryBuilder.createPositionalParameter() );
43      return builder.toString();
44    }

46    /*
