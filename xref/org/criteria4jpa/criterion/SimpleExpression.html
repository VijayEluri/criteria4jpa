23    /**
24     * Creates a simple expression of the form:
25     * <pre> 
26     * &lt;path expression&gt; &lt;operator&gt; &lt;value&gt;
27     * </pre> 
28     * 
29     * @param relativePath relative path of a <i>persistent field</i> or
30     *   <i>single-valued relationship field</i>.
31     * @param value a value to check the field against
32     * @param op the operator
33     */
34    public SimpleExpression(String relativePath, Object value, String op) {
35      this.relativePath = relativePath;
36      this.value = value;
37      this.op = op;
38    }
39  
40    /*
41     * @see org.criteria4jpa.criterion.Criterion#toQueryString(org.criteria4jpa.Criteria, org.criteria4jpa.impl.CriteriaQueryBuilder)
42     */
43    public String toQueryString(Criteria criteria, CriteriaQueryBuilder queryBuilder) {
44      StringBuilder builder = new StringBuilder();
45      builder.append( queryBuilder.getAbsolutePath(criteria, relativePath) );
46      builder.append( ' ' );
47      builder.append( op );
48      builder.append( ' ' );
49      builder.append( queryBuilder.createPositionalParameter() );
50      return builder.toString();
51    }
