22    /**
23     * Creates a custom JPQL restriction.
24     * 
25     * @param jpql The JPQL fragment
26     * @param values The query parameters to add to the query
27     */
28    public JPQLExpression(String jpql, Object[] values) {
29      this.jpql = jpql;
30      this.values = values;
31    }

33    /*
34     * @see org.criteria4jpa.criterion.Criterion#toQueryString(org.criteria4jpa.Criteria, org.criteria4jpa.impl.CriteriaQueryBuilder)
35     */
36    public String toQueryString(Criteria criteria, CriteriaQueryBuilder queryBuilder) {
37      
38      // replace alias variable by real name
39      String queryFragment = 
40        jpql.replace( "{alias}", queryBuilder.getRequiredAlias(criteria) );
41  
42      // process parameters
43      for(int i = 1; i <= values.length; i++) {
44        
45        // create positional parameter for this value
46        String paramater = queryBuilder.createPositionalParameter();
47        
48        /*
49         * Replace variable with generated positional parameter.
50         * Note: This kind of string processing is not very efficient, 
51         * because it generates a new string object in each iteration.
52         * But we assume that the number of iterations will be
53         * relative small, so it won't be a real problem.
54         */
55        queryFragment = queryFragment.replace( "{"+i+"}", paramater);
56        
57      }
58      
59      return queryFragment;
60    }
61
